Pt.1 - Get element by using attribute name

Specify an attribute name inside of [] to choose element by specifying an attribute name.

JS

        
        ////Pt1 - Get element by using attribute name
        let $selectedAttrName= [];
        
        //write selector here
        $selectedAttrName = $("[data-name]");
        
        //Add css to element which matched as ".bingo-AttrName"(color will change to red)
        $selectedAttrName.each(function(index){
            $(this).addClass("bingo-AttrName");
        });
        $(".count-AttrName").text($selectedAttrName.length);

Result :

*The color before is Yellow.The element color which has specified attribute will change.

Numbers of element which has been selected :

1. Steve Reich - 81
2. Ryuichi Sakamoto - 31
3. Brian Eno - 70
4. Steve Roach - 63
5. Alva Noto - 53
6. Aphex Twin - 47

Pt.2 - Get element which is matching with specified attribute name

Use [] with specified attribute name. attribute name inside of [] should be surround by ''.

JS

        
        let $selectedSpecifiedAttrName = [];
        
        //write selector here
        $selectedSpecifiedAttrName = $("[data-age=81]");
        
        //Add css to element which matched as ".bingo-SpecifiedAttrName"(color will change to red)
        $selectedSpecifiedAttrName.each(function (index) {
            $(this).addClass("bingo-SpecifiedAttrName");
        });
        $(".count-SpecifiedAttrName").text($selectedSpecifiedAttrName.length);

Result :

*The color before is Yellow.The element color which has specified attribute will change.

Numbers of element which has been selected :

1. Steve Reich - 81
2. Ryuichi Sakamoto - 31
3. Brian Eno - 70
4. Steve Roach - 63
5. Alva Noto - 53
6. Aphex Twin - 47

Pt.3 - Get element which isn't matching with specified attribute name

Use != with specified attribute name. attribute name inside of [] should be surround by ' '.

JS

        
        let $selectedNotSpecifiedAttrName = [];
        
        //write selector here
        $selectedNotSpecifiedAttrName = $("[data-japanese!='Miso Ramen']");
        
        //Add css to element which matched as ".bingo-NotSpecifiedAttrName"(color will change to red)
        $selectedNotSpecifiedAttrName.each(function (index) {
            $(this).addClass("bingo-NotSpecifiedAttrName");
        });
        $(".count-NotSpecifiedAttrName").text($selectedNotSpecifiedAttrName.length);

Result :

Numbers of element which has been selected :

The color below won't be changed because all of the other element which isn't selected will be change in this case.

Miso Ramen
Tonkotsu Ramen
Shoyu Ramen
Shio Ramen
Jah-Jah Noodle
Abura soba

Pt.4 - Get element which starts with attribute name of a specified keyword

Use ^= with specified attribute name. attribute name inside of [] should be surround by ' '.

*If you want to get element which ends with a specified keyword, use $= instead of ^=.

JS

        
        ////Pt4 - Get element which is matching with specified keyword of attribute name
        let $selectedSpecifiedKeyWordAttr = [];
        
        $selectedSpecifiedKeyWordAttr = $("[data-fruit ^= 'P']");
        
        $selectedSpecifiedKeyWordAttr.each(function (index) {
            $(this).addClass("bingo-SpecifiedAttrName");
        });
        $(".count-SpecifiedKeyWordAttr").text($selectedSpecifiedKeyWordAttr.length);

Result :

Numbers of element which has been selected :

Strawberry
Dragon Fruit
Passion Fruit
Banana
Pineapple
Blueberry

Pt.5 - Get element which includes a specified keyword in attribute name

Use *= with specified attribute name. attribute name inside of [] should be surround by ' '.

JS

        
        let $selectedAttrKeyWordIncl = [];
        
        $selectedAttrKeyWordIncl = $("[data-curry *= 'e']");
        
        $selectedAttrKeyWordIncl.each(function (index) {
            $(this).addClass("bingo-SpecifiedAttrName");
        });
        $(".count-selectedAttrKeyWordIncl").text($selectedAttrKeyWordIncl.length);

Result :

Numbers of element which has been selected :

Indian curry
Vegetable curry
Chicken curry
Keema curry
Green curry
Dry curry
